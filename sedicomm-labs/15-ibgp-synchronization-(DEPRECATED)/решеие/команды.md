# присутпаем к решению

#### На R3, R2 включаем синхронизацию, это позволит
 - перестать анонсировать маршруты 100.0.0.0 и 200.0.0.0
   доставить которые мы пока не можем из-за R3
 - при этом в bgp table эти маршруты будут удерживаться 
   в R1,R2




#### (1) Делаем на примере R2, на R1 все вкючается зеркально

# исходный конфиг R2 bgp для анонсов ibgp+ebgp
    router bgp 200
    no synchronization
    #ibgp neig config
    neighbor <R1-addr> remote-as 200
    neighbor <R1-addr> update-source Loopback0
    neighbor <R1-addr> next-hop-self
    #ebgp neig config
    neighbor R5 <R5-addr> remote-as 1000
# исходный конфиг R2 ospf для связности внутри AS
    router ospf 1
    network <link-to-R3>/30 area 0

#  Итог - получили проблему показанную в объяснении - R3 не знает куда роутить 100.0.0.0 200.0.0.0


#### (2) Решаем проблему 

# включем bgp synchronization R2,R1
# теперь анонсы EBGP соседям не пойдут пока для 100... и 200... из BGP не будет аналогичного префикса в IGP
    router bgp 200
    synhronization

#  включаем редистрибьют bgp маршрутов в OSPF, чтобы префикс роутов 100.. и 200.. появлися в IGP
    router ospf 1
    redistribute bgp 200





#### (3) Итого

# R3 терерь знает как роутить 100... 200...

# R1, R2 анонсит bgp роуты своим EBGP соседям
#на R5, R6

# из минусов R1,R2 теперь имеют копию bgp table в OSPF и R1,R2 RAM явно будет этому не рада 
# особенно если они получают от соседей full-view
#на R1, R2
show ip bgp 
show ip ospf database
